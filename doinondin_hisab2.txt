<body>

  <h1>দৈনন্দিন আয়-খরচ হিসাব</h1>

  <div class="section">
    <h2>আয়</h2>
    <table id="incomeTable">
      <tr><th>তারিখ</th><th>বিবরণ</th><th>টাকা</th></tr>
    </table>
    <input type="text" id="incomeDesc" placeholder="বিবরণ">
    <input type="number" id="incomeAmount" placeholder="টাকা">
    <button onclick="addIncome()">আয় যোগ করুন</button>
  </div>

  <div class="section">
    <h2>খরচ</h2>
    <table id="expenseTable">
      <tr><th>তারিখ</th><th>বিবরণ</th><th>টাকা</th></tr>
    </table>
    <input type="text" id="expenseDesc" placeholder="বিবরণ">
    <input type="number" id="expenseAmount" placeholder="টাকা">
    <button onclick="addExpense()">খরচ যোগ করুন</button>
  </div>

  <div class="section">
    <h2>সারাংশ</h2>
    <p>মোট আয়: <span id="totalIncome">0</span> টাকা</p>
    <p>মোট খরচ: <span id="totalExpense">0</span> টাকা</p>
    <p><strong>মোট ব্যালেন্স: <span id="balance">0</span> টাকা</strong></p>
  </div>

  <script>
    let totalIncome = 0;
    let totalExpense = 0;

    function addIncome() {
      const desc = document.getElementById("incomeDesc").value;
      const amount = parseFloat(document.getElementById("incomeAmount").value);
      if (!desc || isNaN(amount)) return;

      const table = document.getElementById("incomeTable");
      const row = table.insertRow(-1);
      row.innerHTML = `<td>${new Date().toLocaleDateString()}</td><td>${desc}</td><td>${amount}</td>`;

      totalIncome += amount;
      updateSummary();
    }

    function addExpense() {
      const desc = document.getElementById("expenseDesc").value;
      const amount = parseFloat(document.getElementById("expenseAmount").value);
      if (!desc || isNaN(amount)) return;

      const table = document.getElementById("expenseTable");
      const row = table.insertRow(-1);
      row.innerHTML = `<td>${new Date().toLocaleDateString()}</td><td>${desc}</td><td>${amount}</td>`;

      totalExpense += amount;
      updateSummary();
    }

    function updateSummary() {
      document.getElementById("totalIncome").innerText = totalIncome;
      document.getElementById("totalExpense").innerText = totalExpense;
      document.getElementById("balance").innerText = totalIncome - totalExpense;
    }
  </script>
